1. Пользователи и аутентификация (Go модуль)
-- Таблица пользователей
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    email VARCHAR(255) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    full_name VARCHAR(255),
    phone VARCHAR(20),
    avatar_url TEXT,
    timezone VARCHAR(50) DEFAULT 'Europe/Moscow',
    currency_preference VARCHAR(3) DEFAULT 'RUB',
    date_format VARCHAR(20) DEFAULT 'DD.MM.YYYY',
    financial_start_day INTEGER DEFAULT 1, -- 1-е число месяца для бюджетов
    is_premium BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Сессии и токены
CREATE TABLE user_sessions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    device_type VARCHAR(50),
    device_token TEXT,
    refresh_token TEXT NOT NULL,
    expires_at TIMESTAMP WITH TIME ZONE NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

2. Транзакции и категории (Go модуль)
-- Категории транзакций
CREATE TABLE categories (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    name VARCHAR(100) NOT NULL,
    type VARCHAR(10) NOT NULL CHECK (type IN ('INCOME', 'EXPENSE', 'TRANSFER')),
    color VARCHAR(7), -- HEX цвет #RRGGBB
    icon VARCHAR(50),
    parent_category_id UUID REFERENCES categories(id),
    is_system BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE bank_accounts (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    bank_name VARCHAR(100) NOT NULL,
    account_number VARCHAR(100),
    account_type VARCHAR(50),
    balance DECIMAL(15,2),
    currency VARCHAR(3) DEFAULT 'RUB',
    last_synced_at TIMESTAMP WITH TIME ZONE,
    is_active BOOLEAN DEFAULT TRUE,
    external_account_id VARCHAR(255), -- ID из банковской системы
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Финансовые привычки (детектор привычек)
CREATE TABLE habits (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    habit_type VARCHAR(50) NOT NULL, -- 'MORNING_COFFEE', 'SPONTANEOUS_SHOPPING', 'LUNCH_OUTSIDE' и т.д.
    category_id UUID REFERENCES categories(id) ON DELETE SET NULL,
    average_amount DECIMAL(10,2),
    frequency VARCHAR(20) DEFAULT NULL,
    pattern JSONB, -- паттерн поведения (дни недели, время суток)
    confidence_score DECIMAL(5,4),
    impact_score DECIMAL(5,4), -- влияние на бюджет (0-1)
    last_detected DATE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Транзакции
CREATE TABLE transactions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    amount DECIMAL(15,2) NOT NULL,
    currency VARCHAR(3) DEFAULT 'RUB',
    type VARCHAR(10) NOT NULL CHECK (type IN ('INCOME', 'EXPENSE', 'TRANSFER')),
    category_id UUID REFERENCES categories(id),
    description TEXT,
    original_description TEXT, -- оригинальное описание из банка
    date TIMESTAMP WITH TIME ZONE NOT NULL,
    location TEXT, -- место совершения операции
    payment_method VARCHAR(50), -- карта, наличные и т.д.
    bank_account_id UUID REFERENCES bank_accounts(id), -- идентификатор счета в банке
    is_recurring BOOLEAN DEFAULT FALSE, -- регулярная транзакция
    recurrence_pattern JSONB, -- паттерн повторения
    ml_category_id UUID REFERENCES categories(id) ON DELETE SET NULL, -- категория от ML
    ml_confidence DECIMAL(5,4), -- уверенность ML модели
    is_habit_related BOOLEAN DEFAULT FALSE,
    habit_pattern_id UUID REFERENCES habits(id) ON DELETE SET NULL, -- ссылка на habits.id
    tags JSONB, -- пользовательские теги
    is_essential BOOLEAN DEFAULT FALSE, -- обязательный расход
    is_verified BOOLEAN DEFAULT TRUE, -- подтверждена пользователем
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Импорты файлов
CREATE TABLE imports (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    file_name VARCHAR(255) NOT NULL,
    file_type VARCHAR(10) CHECK (file_type IN ('CSV', 'EXCEL', 'JSON')),
    status VARCHAR(20) CHECK (status IN ('PENDING', 'PROCESSING', 'COMPLETED', 'FAILED')),
    total_records INTEGER,
    processed_records INTEGER DEFAULT 0,
    errors JSONB, -- список ошибок при обработке
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

3. Подписки и привычки (Go модуль)
-- Подписки и регулярные платежи
CREATE TABLE subscriptions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    name VARCHAR(200) NOT NULL,
    amount DECIMAL(10,2) NOT NULL,
    currency VARCHAR(3) DEFAULT 'RUB',
    category_id UUID REFERENCES categories(id),
    recurrence VARCHAR(20) NOT NULL CHECK (recurrence IN ('DAILY', 'WEEKLY', 'MONTHLY', 'YEARLY')),
    next_billing_date DATE NOT NULL,
    is_active BOOLEAN DEFAULT TRUE,
    ml_detected BOOLEAN DEFAULT FALSE, -- обнаружено ML
    usage_index DECIMAL(5,4), -- индекс использования (0-1)
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

4. Бюджеты (Java модуль)
-- Бюджеты по категориям
CREATE TABLE budgets (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    category_id UUID REFERENCES categories(id),
    amount_limit DECIMAL(15,2) NOT NULL,
    spent_amount DECIMAL(15,2) DEFAULT 0,
    period VARCHAR(20) NOT NULL CHECK (period IN ('DAILY', 'WEEKLY', 'MONTHLY', 'YEARLY')),
    period_start DATE NOT NULL,
    period_end DATE NOT NULL,
    currency VARCHAR(3) DEFAULT 'RUB',
    alert_thresholds JSONB, -- пороги уведомлений [50, 80, 90, 100]
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Общие бюджеты (без категории)
CREATE TABLE general_budgets (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    total_limit DECIMAL(15,2) NOT NULL,
    spent_amount DECIMAL(15,2) DEFAULT 0,
    period VARCHAR(20) NOT NULL CHECK (period IN ('WEEKLY', 'MONTHLY', 'YEARLY')),
    period_start DATE NOT NULL,
    period_end DATE NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

5. Цели и планирование (Java модуль)
-- Финансовые цели
CREATE TABLE goals (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    name VARCHAR(200) NOT NULL,
    description TEXT,
    target_amount DECIMAL(15,2) NOT NULL,
    current_amount DECIMAL(15,2) DEFAULT 0,
    currency VARCHAR(3) DEFAULT 'RUB',
    deadline DATE NOT NULL,
    goal_type VARCHAR(20) NOT NULL CHECK (goal_type IN ('SAVING', 'DEBT_REPAYMENT', 'INVESTMENT', 'PURCHASE')),
    priority INTEGER DEFAULT 1, -- 1-5 приоритет
    status VARCHAR(20) DEFAULT 'ACTIVE' CHECK (status IN ('ACTIVE', 'COMPLETED', 'CANCELLED')),
    auto_save_amount DECIMAL(10,2), -- сумма автопополнения
    auto_save_frequency VARCHAR(20), -- частота автопополнения
    icon VARCHAR(50),
    color VARCHAR(7),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Транзакции для целей (пополнения)
CREATE TABLE goal_transactions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    goal_id UUID NOT NULL REFERENCES goals(id) ON DELETE CASCADE,
    transaction_id UUID REFERENCES transactions(id),
    amount DECIMAL(10,2) NOT NULL,
    date DATE NOT NULL,
    is_auto_save BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

6. Уведомления (оба модуля)
-- Уведомления (общая таблица)
CREATE TABLE notifications (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    type VARCHAR(50) NOT NULL, -- 'BUDGET_ALERT', 'GOAL_PROGRESS', 'SUBSCRIPTION_REMINDER', 'HABIT_DETECTED', 'LARGE_TRANSACTION', 'SYSTEM'
    title VARCHAR(255) NOT NULL,
    message TEXT NOT NULL,
    source_module VARCHAR(20) NOT NULL CHECK (source_module IN ('GO', 'JAVA', 'ML', 'SYSTEM')),
    entity_type VARCHAR(50), -- 'transaction', 'budget', 'goal', 'subscription'
    entity_id UUID, -- ID связанной сущности
    data JSONB, -- дополнительные данные
    is_read BOOLEAN DEFAULT FALSE,
    is_archived BOOLEAN DEFAULT FALSE,
    scheduled_for TIMESTAMP WITH TIME ZONE, -- для отложенных уведомлений
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Шаблоны уведомлений
CREATE TABLE notification_templates (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    type VARCHAR(50) UNIQUE NOT NULL,
    title_template TEXT NOT NULL,
    message_template TEXT NOT NULL,
    conditions JSONB, -- условия срабатывания
    priority INTEGER DEFAULT 1
);

CREATE TABLE reports (

    widget_type VARCHAR(50) NOT NULL, -- 'SPENDING_CHART', 'BUDGET_STATUS', 'GOAL_PROGRESS', 'RECENT_TRANSACTIONS'
    position INTEGER NOT NULL,
    config JSONB, -- конфигурация виджета
    is_visible BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

8. Голосовой ввод и ML (общие)
-- Голосовые транскрипции
CREATE TABLE voice_transcriptions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    audio_file_url TEXT,
    transcribed_text TEXT NOT NULL,
    entities JSONB, -- извлеченные сущности (сумма, категория, место)
    ml_category_id UUID REFERENCES categories(id),
    confidence DECIMAL(5,4),
    status VARCHAR(20) DEFAULT 'PENDING' CHECK (status IN ('PENDING', 'PROCESSED', 'FAILED')),
    transaction_id UUID REFERENCES transactions(id),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Рекомендации от ML
CREATE TABLE recommendations (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    type VARCHAR(50) NOT NULL, -- 'BUDGET_OPTIMIZATION', 'SUBSCRIPTION_CANCELLATION', 'SAVING_TIP', 'HABIT_CHANGE'
    title VARCHAR(255) NOT NULL,
    description TEXT NOT NULL,
    action_items JSONB, -- конкретные действия
    estimated_savings DECIMAL(10,2),
    priority INTEGER DEFAULT 1,
    is_applied BOOLEAN DEFAULT FALSE,
    applied_at TIMESTAMP WITH TIME ZONE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

Оффлайн-режим (синхронизация):

CREATE TABLE sync_queue (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

    entity_type VARCHAR(50) NOT NULL,
    entity_id UUID,
    entity_data JSONB NOT NULL,
    status VARCHAR(20) DEFAULT 'PENDING' CHECK (status IN ('PENDING', 'SYNCED', 'FAILED')),
    retry_count INTEGER DEFAULT 0,
    last_attempt TIMESTAMP WITH TIME ZONE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

Ограничения
-- Constraints для целей
ALTER TABLE goals ADD CONSTRAINT ck_goals_auto_save_frequency 
CHECK (auto_save_frequency IS NULL OR auto_save_frequency IN ('DAILY', 'WEEKLY', 'MONTHLY', 'YEARLY'));

-- Constraints для привычек
ALTER TABLE habits ADD CONSTRAINT ck_habits_frequency 
CHECK (frequency IS NULL OR frequency IN ('DAILY', 'WEEKLY', 'MONTHLY'));

-- Constraints для транзакций и привычек (диапазоны 0-1)
ALTER TABLE transactions ADD CONSTRAINT ck_ml_confidence_range 
CHECK (ml_confidence IS NULL OR (ml_confidence >= 0 AND ml_confidence <= 1));

ALTER TABLE habits ADD CONSTRAINT ck_habits_confidence_score 
CHECK (confidence_score IS NULL OR (confidence_score >= 0 AND confidence_score <= 1));

ALTER TABLE habits ADD CONSTRAINT ck_habits_impact_score 
CHECK (impact_score IS NULL OR (impact_score >= 0 AND impact_score <= 1));

-- Constraints для подписок
ALTER TABLE subscriptions ADD CONSTRAINT ck_subscriptions_usage_index 
CHECK (usage_index IS NULL OR (usage_index >= 0 AND usage_index <= 1));

-- Constraints для голосового ввода
ALTER TABLE voice_transcriptions ADD CONSTRAINT ck_voice_confidence 
CHECK (confidence IS NULL OR (confidence >= 0 AND confidence <= 1));

Функция для триггера
-- Функция для триггера
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = CURRENT_TIMESTAMP;
    RETURN NEW;
END;
$$ language 'plpgsql';
